def experiment(iteration_max=500, exp_num=10, clf=None, classification=False):
    dtimes_exp = []
    for i in range(exp_num):
        env = System(system_h=20, system_w=20, system_fake=0.3, random_status=True)
        agent = RandomWalk([0, 1, 2, 3])
        dtimes = []
        for episode in range(iteration_max):
            sys.stdout.flush()
            env.reset()
            ite_cnt = 0
            while True:
                ite_cnt += 1
                action = agent.choose_action(env.get_availabel_director(classification=classification))
                _, _, done = env.step(action)

                if episode > 0 and ite_cnt % 100 == 0:
                    print("\rExp %2d: Step %5d @ Episode %3d/%3d (%5d/%5d)" % (i, ite_cnt, episode + 1, iteration_max, 
                                                                     dtimes[-1], min(dtimes)), end="")    
                elif episode == 0 and ite_cnt % 100 == 0:
                    print("\rExp %2d: Step %5d @ Episode %3d/%3d (-1/-1)" % (i, ite_cnt, episode + 1, iteration_max), end="")    
                
                if done:
                    break    
            dtimes.append(ite_cnt)
        # end of game
        print('game over\nExpected iterations: %d' % np.mean(dtimes, dtype=int))
        dtimes_exp.append(dtimes)
    return dtimes_exp

def filter_mean(X, num=20):
    X_ = []
    for x in X:
        x_ = np.array([x[i:i+num].mean() for i in range(len(x) - num + 1)])
        X_.append(x_)
    return np.array(X_)
